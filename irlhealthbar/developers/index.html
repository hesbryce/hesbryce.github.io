Fetch with ETag Caching

// Simple fetch const shareCode = 'ABC12345'; const response = await fetch(`https://stamina-api.onrender.com/${shareCode}`); const data = await response.json(); console.log(`Stamina: ${data.stamina_score}%`); console.log(`Color: ${data.color}`); console.log(`Realtime: ${data.realtime}`); // With ETag caching for efficient polling let etag = null; async function pollStamina() { const headers = etag ? { 'If-None-Match': etag } : {}; const response = await fetch(`https://stamina-api.onrender.com/${shareCode}`, { headers }); if (response.status === 304) { console.log('Data unchanged (304)'); return; } etag = response.headers.get('etag'); const data = await response.json(); updateUI(data); } // Poll based on cache TTL setInterval(pollStamina, data.cache_ttl_seconds * 1000);

import requests import time share_code = "ABC12345" api_url = f"https://stamina-api.onrender.com/{share_code}" etag = None def fetch_stamina(): global etag headers = {'If-None-Match': etag} if etag else {} response = requests.get(api_url, headers=headers) if response.status_code == 304: print("Data unchanged (304)") return None etag = response.headers.get('ETag') data = response.json() print(f"Stamina: {data['stamina_score']}%") print(f"Color: {data['color']}") print(f"Timestamp: {data['timestamp_cst']}") return data # Fetch once data = fetch_stamina() # Poll based on update interval if data: while True: time.sleep(data['update_interval_seconds']) fetch_stamina()
